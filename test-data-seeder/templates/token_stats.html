🔌 How to Integrate
Add to any WorkHive Python file Python 
# 1. Import the tracker and router
from tools.claude_token_tracker import TrackedClient
from tools.claude_model_router  import ModelRouter

router = ModelRouter()
client = TrackedClient()

# 2. Router picks the cheapest model based on your data history
model = router.suggest("search")     # → "claude-haiku-4-5"
model = router.suggest("debugging")  # → "claude-opus-4-7"

# 3. Make the call — it's logged automatically
response = client.chat(
    "Fix the null-pointer in inventory.py",
    model=model,
    action_type="debugging",
)
text = response.content[0].text
